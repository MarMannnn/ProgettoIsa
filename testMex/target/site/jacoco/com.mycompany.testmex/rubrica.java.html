    public String[] lstUtn(){
    String[] utenti = new String[20];
    try{  
        // create the mysql insert preparedstatement
            String query = "SELECT userName FROM utenti";
            PreparedStatement ps = MyConnection.getConnection().prepareStatement(query);          
            ResultSet rs=ps.executeQuery();
            int i = 0;
            while(rs.next()){
            String nome = rs.getString("userName");
            if(!nome.equals(nomeUtente)) {utenti[i] = nome;}
            i++;
            }
            }
        catch(Exception e){ System.out.println(e);} 
    
    return utenti;
    
    }
    
    public String[] lstMsg(){
    String[] messaggi = new String[20];
    
    try{  
        // create the mysql insert preparedstatement
            String query = "SELECT testo,mittente FROM messaggi WHERE destinatario =?";
            PreparedStatement ps = MyConnection.getConnection().prepareStatement(query); 
            ps.setString(1,nomeUtente);
            ResultSet rs=ps.executeQuery();
            int i = 0;
            while(rs.next()){
            String nome = rs.getString("testo");
            String mit = rs.getString("mittente");
            messaggi[i] = mit+": "+nome;
            //messaggi[i] = nome;
            i++;
            
            }
            }
        catch(Exception e){ System.out.println(e);} 
    
    return messaggi;
    
    }
